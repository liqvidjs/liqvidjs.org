The next generation of our [CodeMirror](https://codemirror.net/) plugins is here. They now support recording/replaying multiple files (e.g. `index.html`, `style.css`, and `script.js`), and are much easier to configure. Even better, they're also compatible with other animation libraries like [Remotion](https://www.remotion.dev/) and [GSAP](https://greensock.com/gsap/)!

Check out the demos:

export function Demo() {
  const [url, setUrl] = React.useState("liqvid/codebooth-html");
  // 
  const demos = [
    {title: "HTML", str: "codebooth-html"},
    {title: "Python", str: "codebooth-python"},
    {title: "TSX", str: "codebooth-tsx"},
    {title: "Cursor", str: "cursor"}
  ];
  // 
  const libs = [
    {title: "Liqvid", str: "liqvid", working: ["cursor", "codebooth-html", "codebooth-python"]},
    {title: "Remotion", str: "remotion", working: ["codebooth-html", "codebooth-python", "codebooth-tsx", "cursor"]},
    {title: "GSAP", str: "gsap", working: ["cursor", "codebooth-python", "codebooth-html", "codebooth-tsx"]},
  ];
  //
  const onClick = React.useCallback((e) => {
    const {url} = e.currentTarget.dataset;
    setUrl(url);
  }, []);
  //
  return (
    <div className="demo-matrix">
      <table>
        <caption>Demos</caption>
        <thead>
          <tr>
            <th>Plugin / Library</th>
            {libs.map(lib =>
              <th scope="col" key={lib.str}>{lib.title}</th>
            )}
          </tr>
        </thead>
        <tbody>
          {demos.map(demo => (
            <tr key={demo.str}>
              <th scope="row">{demo.title}</th>
              {libs.map(lib =>
                <td key={lib.str}>
                  <a role="button" onClick={onClick} data-url={`${lib.str}/${demo.str}`}>Demo</a>
                </td>
              )}
            </tr>
          ))}
        </tbody>
      </table>
      <iframe allow="fullscreen" src={`/r/lqv-demos/${url}/`} />
    </div>
  );
}
